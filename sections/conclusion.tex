\begin{frame}
    \begin{center}
        \Large{\textbf{Take-home messages}}
    \end{center}
    ~\\
    \begin{itemize}[label=\textbullet]
        \item Although NP-hard, $\ell_0$-problems are relevant for \textcolor{TolLightOrange}{many applications}
        \item There exists \textcolor{TolLightOrange}{MIP methods} to tackle them
        \begin{itemize}[label=\textbullet]
        \item Generic MIP solvers
        \item Specialized BnB algorithms (\textcolor{TolLightOrange}{\texttt{el0ps}})
        \item Structure-exploitation is key
        \end{itemize}
        \item It's an active research area
        \begin{itemize}[label=\textbullet]
        \item Theoretical/methodological developments still needed
        \item Help us diffusing to practitioners :)
        \end{itemize}
    \end{itemize}
\end{frame}
